1. Типы данных в Python

Python имеет несколько встроенных типов данных. Каждый тип предназначен для хранения определённого вида информации.

int — целые числа

# Целые числа
x = 42
y = -7
print(type(x))  # <class 'int'>

float — числа с плавающей точкой

pi = 3.14159
temperature = -0.5
print(type(pi))  # <class 'float'>

str — строки

name = 'Алексей'
greeting = "Привет, мир!"
print(type(name))  # <class 'str'>

bool — логический тип

is_student = True
has_car = False
print(type(is_student))  # <class 'bool'>

list — список (изменяемый)

fruits = ['яблоко', 'банан', 'апельсин']
numbers = [1, 2, 3, 4, 5]
print(type(fruits))  # <class 'list'>

tuple — кортеж (неизменяемый)

coordinates = (10, 20)
colors = ('красный', 'зелёный', 'синий')
print(type(coordinates))  # <class 'tuple'>

dict — словарь

person = {'имя': 'Иван', 'возраст': 25}
print(type(person))  # <class 'dict'>

set — множество (уникальные элементы)

unique = {1, 2, 3, 2, 1}
print(unique)   # {1, 2, 3}
print(type(unique))  # <class 'set'>

Функция type()

print(type(42))        # <class 'int'>
print(type(3.14))      # <class 'float'>
print(type('hello'))  # <class 'str'>
print(type(True))      # <class 'bool'>

Полный пример (задание ex00)

def data_types():
    a = 10
    b = 'hello'
    c = 3.14
    d = True
    e = [1, 2, 3]
    f = {'key': 'val'}
    g = (1, 2)
    h = {1, 2, 3}
    print([type(a), type(b), type(c), type(d),
           type(e), type(f), type(g), type(h)])

if __name__ == '__main__':
    data_types()

2. Работа с файлами

Python позволяет легко читать и записывать файлы. Основная функция — open().

Режимы открытия файла

'r'   # чтение (файл должен существовать)
'w'   # запись (создаёт или перезаписывает)
'a'   # дозапись в конец
'r+'  # чтение и запись

Конструкция with open() — правильный способ

with open('file.txt', 'r') as f:
    content = f.read()
    print(content)
# Файл закрывается автоматически!
💡 Конструкция with автоматически закрывает файл, даже если возникла ошибка.

Чтение файла

# Весь файл сразу
with open('data.txt', 'r') as f:
    text = f.read()

# Построчно — readlines()
with open('data.txt', 'r') as f:
    lines = f.readlines()  # список строк
    for line in lines:
        print(line.strip())  # убираем \n

Запись в файл

with open('result.txt', 'w') as f:
    f.write('Первая строка\n')
    f.write('Вторая строка\n')

Пример: CSV → TSV (задание ex01)

def convert_csv_to_tsv():
    with open('data.csv', 'r') as csv_file:
        with open('data.tsv', 'w') as tsv_file:
            for line in csv_file:
                parts = []
                in_quotes = False
                current = ''
                for char in line:
                    if char == '"':
                        in_quotes = not in_quotes
                    elif char == ',' and not in_quotes:
                        parts.append(current)
                        current = ''
                        continue
                    current += char
                parts.append(current)
                tsv_file.write('\t'.join(parts))

3. Словари (dict)

Словарь хранит пары ключ: значение. Ключи должны быть уникальными.

Создание и доступ

person = {
    'имя': 'Иван',
    'возраст': 25,
    'город': 'Москва'
}
print(person['имя'])      # Иван
print(person.get('страна', 'РФ'))  # РФ (значение по умолчанию)

Перебор словаря

for key, value in person.items():
    print(f'{key}: {value}')

Проверка наличия ключа

if 'имя' in person:
    print('Ключ существует')

if 'телефон' not in person:
    print('Телефон не найден')

Словарь из списка кортежей

data = [('Россия', '25'), ('Франция', '132'), ('Германия', '132')]
result = {}
for country, number in data:
    if number not in result:
        result[number] = []
    result[number].append(country)

for key, values in result.items():
    for val in values:
        print(f'{key!r} : {val!r}')

Сортировка словаря

data = {'США': 610, 'Испания': 178, 'Франция': 132, 'Германия': 132}

sorted_items = sorted(
    data.items(),
    key=lambda x: (-x[1], x[0])  # убывание по числу, возрастание по имени
)
for country, _ in sorted_items:
    print(country)

4. Аргументы командной строки — sys.argv

Модуль sys позволяет получать аргументы, переданные скрипту при запуске.

Как работает sys.argv

import sys

# Запуск: python3 script.py hello world
# sys.argv = ['script.py', 'hello', 'world']

print(sys.argv[0])   # script.py
print(sys.argv[1])   # hello
print(sys.argv[2])   # world
print(len(sys.argv)) # 3

Проверка количества аргументов

import sys

def main():
    if len(sys.argv) != 2:
        print('Нужен ровно 1 аргумент')
        return
    arg = sys.argv[1]
    print(f'Аргумент: {arg}')

if __name__ == '__main__':
    main()

Вызов исключений

# Вызов исключения с сообщением
if len(sys.argv) != 2:
    raise Exception('Неверное число аргументов!')

# Другой пример:
raise Exception('The script does not support your language yet.')

Преобразование аргументов

# Аргументы всегда строки! Для числа нужно int()
shift = int(sys.argv[3])  # '12' → 12

5. Множества (set)

Множество — коллекция уникальных элементов без определённого порядка. Идеально для математических операций.

Создание множества

# Из списка — дубликаты удаляются
emails = ['a@b.com', 'c@d.com', 'a@b.com']
unique = set(emails)
print(unique)  # {'a@b.com', 'c@d.com'}

# Напрямую
s = {1, 2, 3}

Разность A - B (есть в A, нет в B)

clients = {'ivan@mail.ru', 'anna@mail.ru', 'petr@mail.ru'}
recipients = {'ivan@mail.ru', 'anna@mail.ru'}

not_received = clients - recipients
print(not_received)  # {'petr@mail.ru'}

Пересечение A & B (есть в обоих)

clients = {'ivan@mail.ru', 'anna@mail.ru'}
participants = {'anna@mail.ru', 'new@mail.ru'}

both = clients & participants
print(both)  # {'anna@mail.ru'}

Объединение A | B

all_emails = clients | participants
print(all_emails)  # {'ivan@mail.ru', 'anna@mail.ru', 'new@mail.ru'}

Пример из задания ex07

clients = ['andrew@gmail.com', 'jessica@gmail.com', 'ted@mosby.com']
participants = ['walter@heisenberg.com', 'jessica@gmail.com']
recipients = ['andrew@gmail.com', 'jessica@gmail.com']

# 1. Клиенты без письма → колл-центр
call_center = set(clients) - set(recipients)

# 2. Участники, не клиенты
potential = set(participants) - set(clients)

# 3. Клиенты без участия в событии
loyalty = set(clients) - set(participants)

6. Работа со строками

Строки в Python очень мощные. Они поддерживают десятки методов.

Основные методы строк

s = '  Привет, Мир!  '

s.strip()                    # 'Привет, Мир!'
s.lower()                    # 'привет, мир!'
s.upper()                    # 'ПРИВЕТ, МИР!'
s.capitalize()               # 'Привет, мир!'
s.split(',')                 # ['  Привет', ' Мир!  ']
s.replace('Мир', 'Python')  # замена
s.isalpha()                  # True/False — только буквы
s.isdigit()                  # True/False — только цифры
s.isascii()                  # True/False — только ASCII

f-строки (используйте вместо .format()!)

name = 'Иван'
age = 25

print(f'Меня зовут {name}, мне {age} лет.')
print(f'Через 5 лет мне будет {age + 5} лет.')

price = 1234.567
print(f'Цена: {price:.2f} руб.')  # 1234.57
💡 f-строки быстрее и удобнее. Конструкция '...{}'.format(x) — запрещена в ex08!

Разбивка email (задание ex08)

email = 'ivan.petrov@corp.com'

local = email.split('@')[0]    # 'ivan.petrov'
parts = local.split('.')        # ['ivan', 'petrov']

first = parts[0].capitalize()  # 'Ivan'
last  = parts[1].capitalize()  # 'Petrov'

print(f'{first}\t{last}\t{email}')

Шифр Цезаря — ord() и chr()

print(ord('a'))  # 97
print(ord('z'))  # 122
print(ord('A'))  # 65
print(chr(97))   # 'a'

def shift_char(c, shift):
    if c.islower():
        return chr((ord(c) - ord('a') + shift) % 26 + ord('a'))
    elif c.isupper():
        return chr((ord(c) - ord('A') + shift) % 26 + ord('A'))
    else:
        return c  # цифры, пробелы, символы — без изменений
